3. CBS SERVICE src/modules/cbs/cbs.service.js:

  createGroup(tenantId, data, createdBy):
    1. Save group
    2. Notify assigned leaderId + supervisorId
    3. Return group

  addProspect(tenantId, groupId, data, createdBy):
    1. Check phone not already a prospect in this tenant
       If exists: return existing with "already exists" flag
    2. Save prospect with studyStage: "initial_contact"
    3. Increment group.prospectCount
    4. Notify CBS leader of new prospect
    5. Return prospect

  recordSession(tenantId, groupId, data, conductedBy):
    1. Save session
    2. For each attendee in session.attendees:
       - Find prospect by prospectId
       - Increment prospect.studiesAttended
       - Update prospect.lastStudyDate = session.date
       - If prospect.studyStage = "initial_contact" AND studiesAttended >= 1:
         auto-advance to "interested"
       - If prospect.studyStage = "interested" AND studiesAttended >= 3:
         auto-advance to "studying"
    3. For each decision in session.decisions:
       - If decision.decision = "baptism_request":
         auto-advance prospect.studyStage to "baptism_candidate"
         create notification for supervisor/elder:
         "🎉 Baptism request from [name] in [group]"
    4. Return session

  updateStudyStage(tenantId, prospectId, newStage, updatedBy):
    1. Update stage
    2. If stage = "baptism_candidate":
       notify supervisor + head_pastor
    3. If stage = "baptised":
       update prospect.baptismDate = today
       notify head_pastor + supervisor
    4. If stage = "member":
       offer to auto-convert to member (see convertToMember)
    5. Return updated prospect

  convertToMember(tenantId, prospectId, additionalData, convertedBy):
    1. Fetch prospect
    2. Create Member document using prospect data:
       firstName, lastName, phone, email, gender, branch
       membershipStatus: "new_convert"
       membershipDate: today
       salvationDate: prospect.baptismDate
    3. Update prospect:
       convertedToMemberId = new member._id
       convertedAt = now
       studyStage = "member"
    4. Update group.convertedCount++
    5. Create care case for new convert:
       type: "discipleship", auto-assign to CBS supervisor
    6. Send welcome notification to member
    7. Return { prospect, member }

3. STRATEGIC SERVICE src/modules/strategic/strategic.service.js:

  createKPI(tenantId, data, createdBy):
    1. Link to strategic plan + pillar + focus area
    2. Validate targets (q1+q2+q3+q4 should not need to equal annual
       but warn if they total > annual target)
    3. Save KPI with status "not_started"
    4. Return KPI

  recordKPIValue(tenantId, kpiId, { value, period, notes, enteredBy }):
    1. Find KPI
    2. Push to history array: { period, value, enteredBy, enteredAt, notes }
    3. Update currentValue = value, currentPeriod = period,
       lastUpdated = now
    4. Recalculate status:
       Determine which period target to compare against (q1/q2/q3/q4)
       based on current date
       if value >= annualTarget * 1.05: "exceeded"
       else if value >= periodTarget * 0.9: "on_track"
       else if value >= periodTarget * 0.7: "at_risk"
       else: "off_track"
    5. If status changed to "off_track":
       Notify ownerId: "⚠️ KPI [name] is off track — current: X, target: Y"
    6. Return updated KPI

  autoRefreshKPIs(tenantId):
    Find all KPIs where dataSource !== "manual" AND isActive = true
    For each:
      Pull value from the relevant module using autoQuery config:
      
      "auto_attendance": 
        query Attendance + Service collections
        metrics: avgSundayAttendance, totalHeadcount, attendanceGrowthRate
      
      "auto_finance":
        query Transaction collection
        metrics: monthlyIncome, givingGrowthRate, avgGiving
      
      "auto_members":
        query Member collection
        metrics: totalMembers, newMembersThisMonth, retentionRate
      
      "auto_visitors":
        query Visitor collection
        metrics: totalVisitors, conversionRate, followUpRate
      
      "auto_cbs":
        query CBSProspect collection
        metrics: totalProspects, baptismsThisMonth, studyAttendance
      
      "auto_volunteers":
        query Volunteer collection
        metrics: totalVolunteers, avgReliability
      
      "auto_events":
        query Event collection
        metrics: totalEvents, avgAttendanceRate
      
      Call recordKPIValue() with the pulled value
    
    Return: { refreshed: N, errors: N }

  getBalancedScorecard(tenantId):
    Groups all KPIs by pillar:
    {
      plan: { title, visionStatement },
      pillars: [{
        pillarId, pillarName, color, icon,
        kpis: [{
          kpiId, name, currentValue, annualTarget,
          status, unit, progress: (current/target * 100)
        }],
        pillarScore: avg progress across all KPIs in pillar,
        pillarStatus: "on_track"|"at_risk"|"off_track"
        // pillarStatus = worst status among its KPIs

      overallScore: avg of all pillar scores,
      overallStatus: "on_track"|"at_risk"|"off_track",
      summary: {
        on_track: count,
        at_risk: count,
        off_track: count,
        exceeded: count,
        not_started: count
      }
    }
